def rotate(angle, x = 0, y = 0)
  x2 = @r_x - x
  y2 = @r_y - y
  cos = Math.cos(radians(angle))
  sin = Math.sin(radians(angle))      
  self.class.new(
    x2 * cos - y2 * sin + x, 
    x2 * sin + y2 * cos + y,
    @bounds
  )      
end
