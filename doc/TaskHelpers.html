def stop_tasks
  logger.info("STOP TASKS!")
  if @async_type != :sync
    ExecutorsPool.stop('TaskRunnableLater')
    ExecutorsPool.stop('TaskRunnable')
  else
    logger.info("RunningCode in sync mode")
  end
end
